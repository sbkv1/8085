# Add 2 8-bit numbers

MVI C, 00 
LDA 4300
MOV B, A 
LDA 4301
ADD B
JNC LP1
INR C
LP1: STA 4302
MOV A, C
STA 4303
HLT


# Subtract 2 8-bit numbers

MVI C, 00
LDA 4300
MOV B, A 
LAD 4301
SUB B
JNC LP1
INR C
LP1: STA 4302
MOV A, C
STA 4303
HLT 


# Add n 8-bit number 

MVI D, 00
LXI H, 4300
MOV C, M 
INX H 
DCR C 
MOV A, M 
LP2: INX H 
MOV B, M
ADD B
JNC LP1
INR D 
LP1: DCR C
JNZ LP2
STA 4310
MOV A, D 
STA 4311
HLT


# Multiply 2 8-bit numbers

MVI C, 00
LXI H, 3000
LDA 2000
MOV D, A 
MOV A, M 
DCR D 
REP: MOV E, M 
ADD E
JNC JMP
INR C
JMP: DCR D 
JNZ REP
STA 4000
MOV B, C
STA 4001
HLT


# Division of 2 8-bit numbers

LDA C 2000
MOV C, A
LDA 2001
MVI B, 00
LP2: CMP C 
JC LP1 
SUB C 
INR B
JMP LP2
LP1: STA 3000
MOV A, B
STA 3001
HLT


# Factorial of a 8-bit number

LDA 3000
MOV C, A 
DCR C 
OCR C 
LP1: MOV D, C 
DCR D 
MOV B, A 
LP2: ADD B 
DCR D 
JNZ LP2 
DCR C 
JNZ LP1 
STA 4000
HLT


# Fibonacci series

MVI E, 00
LDA 2000
LXI H, 3000
MOV D, A 
MVI A, 01
MOV M, A 
INX H 
DCR D 
MVI B, 01
MOV M, B
INX H 
DCR D 
LP1: ADD B
MOV M, A 
INX H 
MOV C, A 
MOV A, B
MOV B, C 
DCR D 
JC LP2 
JNZ LP1
LP1: INR E
HLT 


# Smallest Number

LXI H, 2000
MOV C, M 
INX H 
MOV A, M 
DCR C 
INX H 
LOOP: CMP M 
JC SKIP 
MOV A, M 
SKIP: INX H 
DCR C 
JNZ LOOP 
STA 4000
HLT
